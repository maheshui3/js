const movies = [ {title : 'a', year: 2018, rating: 4.1}, {title : 'b', year: 2018, rating: 4.9}, {title : 'c', year: 2018, rating: 3}, {title : 'd', year: 2018, rating: 4.3}, {title : 'a', year: 2018, rating: 4.7} ] //all the movies in 2018 with rating >4 //sort them by rating //descending order based on title console.log(movieFiltering(movies)); function movieFiltering(movies){ return movies.filter(m => m.year === 2018 && m.rating > 4) .sort((a, b) => a.rating - b.rating) .reverse().map(m => m.title); }
