/*Delaying decorator Create a decorator delay(f, ms) that delays each call of f by ms milliseconds In other words, delay(f, ms) returns a "delayed by ms " variant of f*/ function f(x) { console.log(x); } function delay(f, ms) { return function() { setTimeout(() => f.apply(this, arguments), ms); }; } // create wrappers let f1000 = delay(f, 1000); let f1500 = delay(f, 1500); f1000("test"); // shows "test" after 1000ms f1500("test"); // shows "test" after 1500ms
