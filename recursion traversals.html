let company = { sales: [{name: 'John', salary: 1000}, {name: 'Alice', salary: 1600 }], development: { sites: [{name: 'Peter', salary: 2000}, {name: 'Alex', salary: 1800 }], internals: [{name: 'Jack', salary: 1300}] } }; // The function to do the job function sumSalaries(company) { if(Array.isArray(company)){ return company.reduce((acc, cur) => (acc + cur.salary), 0); } else { let sum = 0; for (const subdeep of Object.values(company)) { sum += sumSalaries(subdeep); } return sum; } } console.log(sumSalaries(company));
