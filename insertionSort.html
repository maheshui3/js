/*Bubble sort Insertion Sort is the simplest sorting algorithm, here The array is virtually split into a sorted and an unsorted part. Values from the unsorted part are picked and placed at the correct position in the sorted part. time complexity = O(n^2) //worst time complexity */ /*COMMENTS: */ function insertionSort(arr) { let j, current; for (let i = 1; i < arr.length; i++) { j = i - 1; current = arr[i]; while (j >= 0 && arr[j] > current) { arr[j+1] = arr[j]; j--; } arr[j+1] = current; } return arr; } console.log(insertionSort([1,8,4,2,7])); //comment watch that video https://www.youtube.com/watch?v=yCxV0kBpA6M
