/*write a function called minSubarrayLen which accepts two parameters - array of integers and a positive integer. this function should return minimal length of a contiguous subarray of which the sum is greater than or equal to integer passed to the function, if there isn't one, return 0 instead.*/ function minSubarrayLen(arr, num) { let sum = 0; let start = 0; let end = 0; let minLen = Infinity; for (let end = 0; end < arr.length; end++) { sum += arr[end]; while(sum >= num && start <= end) { minLen = Math.min(minLen, end - start + 1); sum -= arr[start++]; } } return minLen === Infinity ? 0 : minLen; } console.log(minSubarrayLen([1,2,2,3,3,4], 6)); console.log(minSubarrayLen([2,1,6,5,4], 9)); console.log(minSubarrayLen([3,1,7,11,2,9,8,21,62], 52));
