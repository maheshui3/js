/*find the Length of the longest substring without repeating characters*/ function findLongestSubstring(str) { let start = 0; let maxLength = 0; let lookUp = {}; for (let index = 0; index < str.length; index++) { let char = str[index]; if(lookUp[char]) { start = Math.max(start, lookUp[char]); } maxLength = Math.max(maxLength, index-start+1); // store the index of the next char so as to not double count lookUp[char] = index + 1; } return maxLength; } console.log(findLongestSubstring("thisisawesome")); console.log(findLongestSubstring("maheshreddy"));
